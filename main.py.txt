from telegram.ext import Updater, MessageHandler, Filters

TOKEN = "BU_YERGA_BOT_TOKEN_QOY"

# faqat sen yozadigan ID
ADMIN_ID = 123456789

def handle_message(update, context):
    user_id = update.message.from_user.id

    # faqat admin post qiladi
    if user_id == ADMIN_ID:
        if update.message.photo:
            photo = update.message.photo[-1].file_id
            context.bot.send_photo(chat_id=update.effective_chat.id, photo=photo, caption="🏠 Yangi uy")
        elif update.message.video:
            video = update.message.video.file_id
            context.bot.send_video(chat_id=update.effective_chat.id, video=video, caption="🏠 Yangi uy video")
        elif update.message.text:
            context.bot.send_message(chat_id=update.effective_chat.id, text=update.message.text)

    else:
        update.message.reply_text("⛔ Siz faqat ko‘ra olasiz, yozish mumkin emas.")

def main():
    updater = Updater(TOKEN, use_context=True)
    dp = updater.dispatcher

    dp.add_handler(MessageHandler(Filters.all, handle_message))

    updater.start_polling()
    updater.idle()
